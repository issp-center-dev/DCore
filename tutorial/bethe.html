Tutorial with t2g Bethe lattice¶

dmft_bethe.ini

[model]
t = 1.0
U = 8.0
J = 1.333333333333333
lattice = bethe
seedname = bethe
orbital_model = t2g
nelec = 1.6

[system]
nk = 100
beta = 50.0

[impurity_solver]
name = TRIQS/cthyb
n_cycles{int} = 500000

[control]
max_step = 40

dcore_pre¶

$ pydmf_pre dmft_bethe.ini

dcore¶

$ pydmf dmft_bethe.ini

dcore_post¶

$ pydmf_post dmft_bethe.ini
