Tutorial with single-band 2D Hubbard model¶

dmft_square.ini

[model]
seedname = square
lattice = square
orbital_model = single
nelec = 1.0
t = -1.0
U = 2.0

[system]
beta = 50.0
nk = 32

[impurity_solver]
name = TRIQS/hubbard-I

[control]
max_step = 10

[tool]
broadening = 0.1
nnode = 4
knode = [(G,0.0,0.0,0.0),(X,0.5,0.0,0.0),(M,0.5,0.5,0.0),(G,0.0,0.0,0.0)]
nk_line = 100
omega_max =5.0
omega_min =-5.0
Nomega = 400

dcore_pre¶

$ pydmf_pre dmft_square.ini

dcore¶

$ pydmf dmft_square.ini

dcore_post¶

$ pydmf_post dmft_square.ini
$ gnuplot square_akw.gp

gnuplot> set xlabel "Energy"
gnuplot> set ylabel "DOS"
gnuplot> plot "square_dos.dat" w l
