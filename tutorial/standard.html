Tutorial with preset model¶

t2g Bethe lattice¶

pydmft_pre¶

model.in

t = 2.0
U = 16.0
J = 2.666666666666667
lattice = bethe
seedname = bethe
model = t2g
nk = 10
nelec = 1.5

$ pydmf_pre model.in

pydmft¶

input.ini

[system]
beta = 100.0

[impurity_solver]
name = TRIQS/hubbard-I

$ pydmf input.ini bethe

pydmft_post¶

input.ini

[system]
beta = 100.0

[impurity_solver]
name = TRIQS/hubbard-I

$ pydmf_post input.ini bethe
